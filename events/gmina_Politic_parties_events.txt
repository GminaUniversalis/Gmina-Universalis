	title = politics.1.t
	desc = politics.1.d

		has_institution = renaissance
		OR = {
			religion = communism_pzpr

			religion = communism_sd
		}
		
		any_owned_province = {

				RELIGION = liberalism_ud
			}
		}
    }

    mean_time_to_happen = {
	    months = 30

		modifier = {
			factor = 0.9
			stability = 1
		}

		modifier = {
			factor = 0.9
			stability = 2
		}

		modifier = {
			factor = 0.9
			stability = 3
		}

		modifier = {
			factor = 0.8
			has_center_of_trade_of_level = 2
		}

		modifier = {
			factor = 0.8
			has_center_of_trade_of_level = 3
		}

		modifier = {
			factor = 0.8
			trade_income_percentage = 0.4
		}

		modifier = {
			factor = 0.8
			trade_income_percentage = 0.5
		}

		modifier = {
			factor = 0.8
			trade_income_percentage = 0.6
		}
	}

	after = {
		hidden_effect = {
			random_country = {
				type = random
				amount = 4
				limit = {
					NOT = {
						tag = ROOT
					}
					OR = {
						religion = communism_pzpr
						religion = communism_zsl
						religion = communism_sd
					}
					any_owned_province = {
						can_have_center_of_reformation_trigger = {
							RELIGION = liberalism_ud
						}
					}
				}

				country_event = { 
					id = politics.101
					days = 30
				}
			}
		}
	}

	option = {
		name = politics.1.a

		ai_chance = { factor = 100 }

		enable_religion = liberalism_ud

		random_owned_province = {
			limit = {
				can_have_center_of_reformation_trigger = {
					RELIGION = liberalism_ud
				}
			}
			change_religion = liberalism_ud
			add_reform_center = liberalism_ud
			add_permanent_province_modifier = {
				name = "religious_zeal_at_conv"
				duration = 9000
			}
			if = {
				limit = {
					NOT = { is_capital = yes }
				}
				root = {
					capital_scope = {
						change_religion = liberalism_ud
						add_permanent_province_modifier = {
							name = "religious_zeal_at_conv"
							duration = 9000
						}
					}
				}
			}
		}

	}

	option = {
		name = politics.1.b

	}
