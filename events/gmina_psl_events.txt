namespace = psl_events

country_event = {
	id = psl_events.1
	title = psl_events.1.t
	desc = psl_events.1.d
	picture = ADVISOR_eventPicture

	trigger = {

		religion = conservatism_psl
		
		1500 = {
			check_variable = {
				which = clock_year
				value = 1991
			}
			
		}
		1500 = {
			check_variable = {
				which = clock_month
				value = 6
			}
		}

		current_guru_is = guru_roman_bartoszcze
		NOT = { has_global_flag = guru_event_happening }
	}

	mean_time_to_happen = {
		days = 14
	}
	
	immediate = {
		set_global_flag = guru_event_happening
	}

	after = {
		clr_global_flag = guru_event_happening
	}

	option = {
		name = psl_events.1.a
		appoint_next_guru = yes
	}
}
