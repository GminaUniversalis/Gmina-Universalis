#Oskarek tu byl

namespace = flavor_pis

normal_or_historical_nations = yes

country_event = {
	id = flavor_pis.1
	title = flavor_pis.1.t
	desc = flavor_pis.1.d
	picture = BUDDHA_eventPicture
	
	is_triggered_only = yes
	
	trigger = {
		FROM = {
			OR = {
				has_province_modifier = emerald_kaczka
				has_province_modifier = emerald_kaczka_alternate
			}
			NOT = {
				owned_by = ROOT
			}
			NOT = {
				owner = {
					OR = {
						is_in_war = {
							defenders = PREV
							defenders = ROOT
						}
						is_in_war = {
							attackers = PREV
							attackers = ROOT
						}
					}
				}
			}
		}
	}
	
	immediate = {
		hidden_effect = {
			FROM = {
				save_global_event_target_as = emerald_kaczka_province
				remove_province_modifier = emerald_kaczka
				remove_province_modifier = emerald_kaczka_alternate
			}
		}
	}
	
	option = {
		name = flavor_pis.1.a
		add_prestige = 10
		hidden_effect = {
			FROM = {
				owner = {
					country_event = {
						id = flavor_pis.2
					}
				}
			}
		}
		custom_tooltip = emerald_kaczka_tt
		capital_scope = {
			add_permanent_province_modifier = {
				name = emerald_kaczka
				duration = -1
			}
		}
	}
}
