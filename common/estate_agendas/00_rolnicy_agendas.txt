		}
	}
	}

	factor = 1
	}

	if = {
		limit = {
			has_estate_agenda_auto_completion = {
				estate = estate_rolnicy
			}
		}
		has_estate_agenda_auto_completion = {
			estate = estate_rolnicy
		}
	}
	else = {
		employed_advisor = {
			category = ADM
		}

	}
	task_completed_effect = {

		estate = estate_rolnicy
	}

		estate = estate_rolnicy
		desc = EST_VAL_AGENDA_DENIED
		loyalty = -5

	}
	}

estate_rolnicy_develop_x = {
	selection_weight = {
		factor = 1
	}
	can_select = {
		capital_scope = {
			is_state_core = root
		}
	}
	pre_effect = {
		random_list = {
			1 = {
				random_owned_province = {
					limit = {
						is_state_core = root
						area_for_scope_province = {
							is_capital_of = root
						}
					}
					save_event_target_as = agenda_province
				}
			}
			2 = {
				trigger = {
					any_owned_province = {
						is_state_core = root
						OR = {
							AND = {
								religion = root
								OR = {
									culture = root
									has_owner_accepted_culture = yes
								}
							}
						}
						region_for_scope_province = {
							is_capital_of = root
						}
					}
				}
				random_owned_province = {
					limit = {
						is_state_core = root
						OR = {
							AND = {
								religion = root
								OR = {
									culture = root
									has_owner_accepted_culture = yes
								}
							}
						}
						region_for_scope_province = {
							is_capital_of = root
						}
					}
					save_event_target_as = agenda_province
				}
			}
			1 = {
				random_owned_province = {
					limit = {
						is_state_core = root
					}
					save_event_target_as = agenda_province
				}
			}
		}
		event_target:agenda_province = {
			if = {
				limit = {
					check_variable = {
						which = province_mil_var
						value = 1
					}
				}
				set_variable = {
					which = province_mil_var
					value = 0
				}
			}
			export_to_variable = {
				which = province_mil_var
				value = trigger_value:base_manpower
			}
			change_variable = {
				which = province_mil_var
				value = 2
			}
		}
	}
	provinces_to_highlight = {
		province_id = event_target:agenda_province
	}
	fail_if = {
		event_target:agenda_province = {
			NOT = { owned_by = root }
		}
	}
    task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_rolnicy
				}
			}
			has_estate_agenda_auto_completion = {
				estate = estate_rolnicy
			}
		}
		else = {
			event_target:agenda_province = {
				base_manpower = "province_mil_var"
			}
		}
    }
    task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_rolnicy
		}
		add_estate_loyalty = {
			estate = estate_rolnicy
			loyalty = 10
		}
		event_target:agenda_province = {
			add_base_manpower = 1
			add_base_production = 1
		}
    }
    failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
		event_target:agenda_province = {
			if = {
				limit = { owned_by = root }
				add_named_unrest = {
					name = local_rolnicy_displeased
					value = 5
				}
			}
		}
        add_estate_loyalty_modifier = {
            estate = estate_rolnicy
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
    }
}

		}

		}

				coalition_target = root
			}
		}
	}

					has_opinion = {

					any_owned_province = { #No need for this agenda if covered by a mission
						OR = {
							is_permanent_claim = root
							is_core = root
						}
					}
				}

					knows_country = prev
				}

					OR = {
						has_terrain = grasslands
					}
					any_neighbor_province = {
						owned_by = root
					}
					NOT = {
						area_for_scope_province = {
							OR = {
								is_empty = yes

							}
						}
					}
				}

					is_part_of_hre = no
					NOT = { hre_reform_passed = landfriede }
					ROOT = { is_part_of_hre = no }
				}
			}
			random_owned_province = {
				limit = {
					OR = {
						has_terrain = grasslands
					}
					any_neighbor_province = {
						owned_by = root
					}
					area_for_scope_province = {
						type = all
						NOT = { owned_by = root }
						is_empty = no
					}
				}
				save_event_target_as = agenda_province
			}
