estate_retirees_hire_advisor = {
	can_select = { 
		NOT = {
			employed_advisor = {
				category = ADM
			}
		}
	}

	selection_weight = {
		factor = 1
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_retirees
				}
			}
			has_estate_agenda_auto_completion = {
				estate = estate_retirees
			}
		}
		else = {
			employed_advisor = {
				category = ADM
			}
			is_in_deficit = no
		}
	}
	task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_retirees
	}
		add_estate_loyalty = {
			estate = estate_retirees
			loyalty = 10
		}
	}
	failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
		add_estate_loyalty_modifier = {
			estate = estate_retirees
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
		}
	}
}
estate_retirees_build_temple_in_y = {
	can_select = {
		adm_tech = 4
		any_owned_province = {
			is_state_core = root
			NOT = { has_building = cathedral }
			OR = {
				AND = {
					NOT = { has_building = temple }
					num_free_building_slots = 1
				}
				AND = {
					ROOT = { adm_tech = 19 }
					has_building = temple
				}
			}
		}
	}
	selection_weight = {
		factor = 1
		modifier = {
			factor = 0.65
			NOT = {
				any_owned_province = {
					is_state_core = root
					base_tax = 4
					NOT = { has_building = cathedral }
					OR = {
						AND = {
							NOT = { has_building = temple }
							num_free_building_slots = 1
						}
						AND = {
							ROOT = { adm_tech = 19 }
							has_building = temple
						}
					}
				}
			}
		}
		modifier = {
			factor = 1.2
			any_owned_province = {
				is_state_core = root
				base_tax = 7
				NOT = { has_building = cathedral }
				OR = {
					AND = {
						NOT = { has_building = temple }
						num_free_building_slots = 1
					}
					AND = {
						ROOT = { adm_tech = 19 }
						has_building = temple
					}
				}
			}
		}
		modifier = {
			factor = 2
			NOT = {
				crown_land_share = 15
			}
		}
	}
	pre_effect = {
		set_variable = {
			which = estate_retirees_build_temple_in_y_var
			value = 0
		}
		random_owned_province = {
			limit = {
				is_state_core = root
				NOT = { has_building = cathedral }
				OR = {
					AND = {
						NOT = { has_building = temple }
						num_free_building_slots = 1
					}
					AND = {
						ROOT = { adm_tech = 19 }
						has_building = temple
					}
				}
			}
			save_event_target_as = agenda_province
		}
		random_owned_province = {
			limit = {
				is_state_core = root
				base_tax = 4
				NOT = { has_building = cathedral }
				OR = {
					AND = {
						NOT = { has_building = temple }
						num_free_building_slots = 1
					}
					AND = {
						ROOT = { adm_tech = 19 }
						has_building = temple
					}
				}
			}
			save_event_target_as = agenda_province
		}
		random_owned_province = {
			limit = {
				is_state_core = root
				base_tax = 7
				NOT = { has_building = cathedral }
				OR = {
					AND = {
						NOT = { has_building = temple }
						num_free_building_slots = 1
					}
					AND = {
						ROOT = { adm_tech = 19 }
						has_building = temple
					}
				}
			}
			save_event_target_as = agenda_province
		}
		event_target:agenda_province = {
			if = {
				limit = {
					NOT = { has_building = temple }
				}
				ROOT = {
					set_variable = {
						which = estate_retirees_build_temple_in_y_var
						value = 1
					}
				}
			}
			else = {
				ROOT = {
					set_variable = {
						which = estate_retirees_build_temple_in_y_var
						value = 2
					}
				}
			}
		}
	}
	provinces_to_highlight = {
		province_id = event_target:agenda_province
	}
	fail_if = {
		event_target:agenda_province = {
			NOT = { owned_by = root }
		}
	}
    task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = {
					estate = estate_retirees
				}
			}
			has_estate_agenda_auto_completion = {
				estate = estate_retirees
			}
		}
		else = {
			if = {
				limit = {
					check_variable = {
						which = estate_retirees_build_temple_in_y_var
						value = 2
					}
				}
				event_target:agenda_province = {
					has_building = cathedral
				}
			}
			else = {
				event_target:agenda_province = {
					OR = {
						has_building = temple
						has_building = cathedral
					}
				}
			}
		}
    }
    task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_retirees
		}
		add_estate_loyalty = {
			estate = estate_retirees
			loyalty = 10
		}
		if = {
			limit = {
				NOT = { crown_land_share = 15 }
				estate_territory = {
					estate = estate_retirees
					territory = 10
				}
				estate_not_has_exempt_from_seize_land_privilege = { estate = estate_retirees }
			}
			take_estate_land_share_small = { estate = estate_retirees }
		}
		else = {
			event_target:agenda_province = {
				add_base_tax = 2
			}
		}
    }
    failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
		event_target:agenda_province = {
			if = {
				limit = { owned_by = root }
				add_named_unrest = {
					name = local_retirees_displeased
					value = 5
				}
			}
		}
		add_estate_loyalty_modifier = {
			estate = estate_retirees
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
		}
    }
}
