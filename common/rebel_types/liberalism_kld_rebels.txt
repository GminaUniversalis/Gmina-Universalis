liberalism_kld_rebels = {
	color = { 170 95 25 }

	area = nation 					# Stays within the nation
	government = theocracy		# Likes monarchy
	defection = none				# Does not defect
	independence = none				# Does not strive for independence
	unit_transfer = no
	gfx_type = culture_owner	
	will_relocate = yes	
	
	resilient = yes
	reinforcing = yes
	general = yes
	smart = yes
	
	artillery = 0.2
	infantry = 0.65
	cavalry = 0.15
	
	morale = 1
	
	# Possible handle actions
	handle_action_negotiate = yes
	handle_action_stability = yes
	handle_action_build_core = yes
	handle_action_send_missionary = yes		
	
	# The rebel type with the highest modifier for this province gets picked
	spawn_chance = {
		factor = 1
		modifier = {
			factor = 0
			has_owner_religion = yes
		}
		modifier = {
			factor = 5
			owner = {
				has_disaster = estate_liberalism_kld_disaster
			}
		}
		modifier = {
			factor = 0.01
			is_overseas = yes
		}
	}
 
 	# This is checked for EACH province in the Area of Operations
	movement_evaluation = {
		factor = 1
		modifier = {
			factor = 0.1
			units_in_province = 1
		}
		modifier = {
			factor = 0.001
			controlled_by = REB
		}
		modifier = {
			factor = 1.5
			unrest = 2
		}
		modifier = {
			factor = 0.0
			religion = liberalism_kld
			owner = { religion = liberalism_kld }
		}
		modifier = {
			factor = 2.0
			NOT = { religion = liberalism_kld }
			any_neighbor_province = {
	 			religion = liberalism_kld
	 			NOT = { has_province_modifier = "forced_conversion" }
	 		}
	 	}
		modifier = {
			factor = 5
			has_missionary = yes
			religion = liberalism_kld
		}
		modifier = {
			factor = 1.5
			unrest = 4
		}
		modifier = {
			factor = 1.5
			unrest = 6
		}
	}
 
 	# Province Scope
 	siege_won_trigger = {
		NOT = { local_autonomy = 50 }
	}
	siege_won_effect = {
		custom_tooltip = religious_rebels_seige_effect_tooltip
		hidden_effect = {
			if = {
				limit = {
					OR = {
						religion = liberalism_kld
						is_reformation_center = yes
						owner = { religion = liberalism_kld }
					}
				}
				add_province_modifier = {
					name = "religious_violence"
					duration = 730
				}
			}
			if = {
				limit = {
					NOT = { religion = liberalism_kld }
					is_reformation_center = no
					NOT = { owner = { religion = liberalism_kldn } }
				}
				change_religion = liberalism_kld
				add_province_modifier = {
					name = "forced_conversion"
					duration = 730
				}
			}
		}
	}
	
	# Country scope
	can_negotiate_trigger = {
		OR = {
			religion_group = liberalism_group
			dominant_religion = liberalism_kld
		}
	}
	
	# Country scope
	can_enforce_trigger = {
		always = yes
	}
	
	# Localisation for their demands
	demands_description = liberalism_kld_rebels_demands
	
	# Country Scope	
	demands_enforced_effect = {
		hidden_effect = {
			add_country_modifier = {
				name = just_lost_to_rebels
				duration = 10
				hidden = yes
			}
		}
		if = {
			limit = {
				NOT = { religion_group = liberalism_group }
			}
			add_prestige = -50
			every_owned_province = {
				limit = {
					has_rebel_faction = REB
				}
				add_local_autonomy = 20
			}
		}
		if = {
			limit = {
				religion = liberalism_kld
			}
			add_country_modifier = {
				name = "religious_intolerance_enforced"
				duration = 3650
			}
		}
		else_if = {
			limit = {
				NOT = { religion = liberalism_kld }
				dominant_religion = liberalism_kld
			}
			change_religion = liberalism_kld
			force_converted = yes
		}
		else_if = {
			limit = {
				NOT = { religion = liberalism_kld }
				NOT = { dominant_religion = liberalism_kld }
				religion_group = liberalism_group
				NOT = { has_country_modifier = heretic_tolerance }
			}
			
			add_country_modifier = {
				name = "heretic_tolerance"
				duration = 3650
			}
		}
		else = { add_stability = -1 } #Should probably never get here
	}
}
