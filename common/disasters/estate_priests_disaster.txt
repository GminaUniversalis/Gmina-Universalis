estate_priests_disaster = {
	potential = {
		has_estate = estate_priests
		estate_influence = {
			estate = estate_priests
			influence = 80
		}
	}


	can_start = {
		has_any_disaster = no
		estate_influence = {
			estate = estate_priests
			influence = 100
		}
		is_at_war = no
		NOT = { government = theocracy }
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			NOT = {
				estate_influence = {
					estate = estate_priests
					influence = 100
				}		
			}
		}
	}
	
	down_progress = {
		factor = 0
		modifier = {
			factor = 1
			NOT = {
				estate_influence = {
					estate = estate_priests
					influence = 100
				}
			}
		}
	}
	
	progress = {
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_priests
					loyalty = 60
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_priests
					loyalty = 50
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_priests
					loyalty = 40
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 1
			NOT = {
				estate_loyalty = {
					estate = estate_priests
					loyalty = 30
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 1
			NOT = {
				estate_loyalty = {
					estate = estate_priests
					loyalty = 20
				}
			}
			is_at_war = no
		}
	}
	
	can_end = {
		custom_trigger_tooltip = {
			tooltip = EST_CRUSHED_PRIESTS
			OR = {
				NOT = { has_country_flag = estate_priests_in_power }
				# has_country_flag = estate_nobles_in_power
				# has_country_flag = estate_church_in_power
			}						
		}
	}
	
	modifier = {	
		global_unrest = 4
		stability_cost_modifier = 0.5
		republican_tradition = -1
		legitimacy = -3
		meritocracy = -3
		devotion = -3
		horde_unity = -3
		tolerance_own = -8
		tolerance_heathen = 8
		global_religious_conversion_resistance = 20
	}

	on_start = gmina_estate_disasters.18
	on_end = gmina_estate_disasters.19
	
	on_monthly = {
	}
}
