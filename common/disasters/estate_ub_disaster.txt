	potential = {
		has_estate = estate_ub
		estate_influence = {
			estate = estate_ub
			influence = 60
		}
	}

		has_any_disaster = no
		estate_influence = {
			estate = estate_ub
			influence = 100
		}
		is_at_war = no
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			NOT = {
				estate_influence = {
					estate = estate_ub
					influence = 100
				}		
			}
		}
	}
	
	down_progress = {
		factor = 0
		modifier = {
			factor = 1
			NOT = {
				estate_influence = {
					estate = estate_ub
					influence = 100
				}
			}
		}
	}
	
	progress = {
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_ub
					loyalty = 60
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_ub
					loyalty = 50
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_ub
					loyalty = 40
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 1
			NOT = {
				estate_loyalty = {
					estate = estate_ub
					loyalty = 30
				}
			}
			is_at_war = no
		}

			factor = 1

				estate_loyalty = {
					estate = estate_ub
					loyalty = 20
				}
			}
			is_at_war = no
		}
	}
	
	can_end = {
		custom_trigger_tooltip = {
			tooltip = EST_CRUSHED_UB
			OR = {
				NOT = { has_country_flag = estate_ub_in_power }
				# has_country_flag = estate_nobles_in_power
				# has_country_flag = estate_church_in_power
			}						
		}
	}
	
	modifier = {	
		interest = 5
		inflation_reduction = 0.5
		production_efficiency = -0.2
		trade_efficiency = -0.2
	}
