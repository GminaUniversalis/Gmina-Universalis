estate_army_disaster = {
	potential = {
		has_estate = estate_army
		estate_influence = {
			estate = estate_army
			influence = 60
		}
	}


	can_start = {
		has_any_disaster = no
		estate_influence = {
			estate = estate_army
			influence = 100
		}
		is_at_war = no
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			NOT = {
				estate_influence = {
					estate = estate_army
					influence = 100
				}		
			}
		}
	}
	
	down_progress = {
		factor = 0
		modifier = {
			factor = 1
			NOT = {
				estate_influence = {
					estate = estate_army
					influence = 100
				}
			}
		}
	}
	
	progress = {
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_army
					loyalty = 60
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_army
					loyalty = 50
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_army
					loyalty = 40
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 1
			NOT = {
				estate_loyalty = {
					estate = estate_army
					loyalty = 30
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 1
			NOT = {
				estate_loyalty = {
					estate = estate_army
					loyalty = 20
				}
			}
			is_at_war = no
		}
	}
	
	can_end = {
		custom_trigger_tooltip = {
			tooltip = EST_CRUSHED_ARMY
			OR = {
				NOT = { has_country_flag = estate_army_in_power }
				# has_country_flag = estate_nobles_in_power
				# has_country_flag = estate_church_in_power
			}						
		}
	}
	
	modifier = {	
		discipline = -0.2
		land_morale = -0.25
		reinforce_speed = -0.5
		infantry_cost = 0.25
	}

	on_start = gmina_estate_disasters.5
	on_end = gmina_estate_disasters.6
	
	on_monthly = {
	}
}
