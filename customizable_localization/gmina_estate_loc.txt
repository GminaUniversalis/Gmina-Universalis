#Miners
defined_text = {
	name = GetMinersAvailability

	text = {
		localisation_key = string_available_miners

		trigger = {
			OR = {
				gu_coal = 5
				capital_scope = {
					OR = {
						region = dolnoslaskie_region
						region = opolskie_region
						region = slaskie_region
					}
				}
			}
		}
	}

	text = {
		localisation_key = string_NOT_available_miners

		trigger = {
			OR = {
				AND = {
					NOT = { gu_coal = 5 }
					capital_scope = {
						OR = {
							region = dolnoslaskie_region
							region = opolskie_region
							region = slaskie_region
						}
					}
				}
				
				AND = {
					NOT = { gu_coal = 5 }
					NOT = {
						capital_scope = {
							OR = {
								region = dolnoslaskie_region
								region = opolskie_region
								region = slaskie_region
							}
						}
					}
				}
			}
		}
	}
}

#Investors
defined_text = {
	name = GetInvestorsAvailability

	text = {
		localisation_key = string_available_investors

		trigger = {

			NOT = { religion_group = communism_group }
			
			OR = {

				colony = 1
				
				AND = {
					gu_finances = 1
					NOT = { current_age = age_of_discovery }
				}
	
				NOT = { 
					OR = {
						current_age = age_of_discovery
						current_age = age_of_reformation
					}
				}
	
			}
		}
	}

	text = {
		localisation_key = string_NOT_available_investors

		trigger = {

			OR = {

				OR = {
					AND = {
						NOT = { gu_finances = 1 }
						current_age = age_of_discovery 
					}
					AND = {
						NOT = { gu_finances = 1 }
						NOT = { current_age = age_of_discovery }
					}
					AND = {
						gu_finances = 1
						current_age = age_of_discovery 
					}

					AND = {
						NOT = { colony = 1 }
						NOT = { gu_finances = 1 }
					}
					AND = {
						NOT = { gu_finances = 1 }
						NOT = { colony = 1 }
						OR = {
							current_age = age_of_discovery 
							current_age = age_of_reformation 
							current_age = age_of_absolutism
							current_age = age_of_revolutions
						}
					}

					OR = {
						current_age = age_of_discovery 
						AND = {
							current_age = age_of_reformation 
							NOT = { gu_finances = 1 }
						}

					}
					
				}

				religion_group = communism_group
	
			}
		}
	}
}
#Farmers
defined_text = {
	name = GetFarmersAvailability

	text = {
		localisation_key = string_available_farmers

		trigger = {
			
			OR = {
				AND = {
					num_of_owned_provinces_with = {
						value = 5
						has_terrain = farmlands
					}
					NOT = { religion_group = communism_group }
				}
				AND = {
					num_of_owned_provinces_with = {
						value = 10
						has_terrain = grasslands
					}
				   NOT = { religion_group = communism_group }
				}

				capital_scope = {
					OR = {
						region = lubelskie_region
						region = wielkopolska_region
						region = kujawsko_pomorskie_region
					}
				}
			}
		}
	}

	text = {
		localisation_key = string_NOT_available_farmers

		trigger = {

			OR = {
				AND = {
					any_owned_province = {
						farmlands = 5 
					}
					religion_group = communism_group
				}
				AND = {
					NOT = {
						any_owned_province = {
							farmlands = 5
						}
					}
					NOT = { religion_group = communism_group }
				}
				AND = {
					any_owned_province = {
						 grasslands = 10
					}
					religion_group = communism_group
				}
				AND = {
					NOT = {
						any_owned_province = {
							grasslands = 10
						}
					}
					NOT = { religion_group = communism_group }
				}
				
				AND = {
					NOT = {
						capital_scope = {
							OR = {
								region = lubelskie_region
								region = wielkopolska_region
								region = kujawsko_pomorskie_region
							}
						}
					
						OR = {
							any_owned_province = {
								grasslands = 10
							}
							any_owned_province = {
								farmlands = 5
							}
						}
					
					}	
				}
				AND = {

					capital_scope = {
						OR = {
							region = lubelskie_region
							region = wielkopolska_region
							region = kujawsko_pomorskie_region
						}
					}

						NOT = {
							OR = {
								any_owned_province = {
									grasslands = 10
								}
								any_owned_province = {
									farmlands = 5
								}

								religion_group = communism_group
							}
						
						}	
				}

			}

		}
	}
}
